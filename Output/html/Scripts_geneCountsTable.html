fitglm <- function(counts, conditions) {
  p.values <- apply(counts, 1, function(row) {
    geneData <- data.frame(count = as.numeric(row), condition = conditions)
    model <- suppressWarnings(glm.nb(count ~ condition, data = geneData))
    conditionFC <- (summary(model) %>% coef)[, 3][2]
    2*(1 - pnorm(conditionFC %>% abs)) %>% as.numeric
  }
  )
  p.values
}
